layer.on('click', function(e) {
    if (e.value !== null) {
        let vector = e.value;
        let v = vector.magnitude().toFixed(2);
        let d = vector.directionTo().toFixed(0);

        let html = (`${v} m/s to ${d}°`);
        let popup = L.popup().setLatLng(e.latlng)
            .setContent(html).openOn(map);
    }
});
* Hay una descarga completa de datos al cliente (`Field`) * El dato está accesible para consulta, con `click` o `mouseover` * Se usa *directionTo* (convenio en corrientes), pero existe también *directionFrom*
